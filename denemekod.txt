#include "stm32f407xx.h"

uint32_t systemClock;

void GPIO_Config(void)
{

	RCC->AHB1ENR |= 1<<3; // GPIOD Clock enable

	GPIOD->MODER |= 1 << 24;   // GPIOD 12.PÄ°N output
	GPIOD->MODER &= ~(1 << 25);

}


int main(void)
{

	GPIO_Config();

	RCC->APB1ENR |= RCC_APB1ENR_TIM3EN;
	TIM3->PSC = 1680;
	TIM3->ARR |= 65535;
	TIM3->CR1 |= TIM_CR1_CEN;

 	while (1)
 	{
 		if(TIM3->CNT > 19320)
 		{
 			TIM3->CR1 &= ~TIM_CR1_CEN;
 			GPIOD->ODR ^= 0x1000;
 			TIM3->CNT = 1;
 			TIM3->CR1 |= TIM_CR1_CEN;
 		}
 	}

}
