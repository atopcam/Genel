    static long ASCIIWordSum(String str, long []sumArr)
    {
        int l = str.Length;
        int pos = 0;
        long top = 0;
        long bigtop = 0;
        for (int x = 0; x < l; x++) {
 
            // her kelimeyi ayir
                  // her kelimeye karsilik gelenler
            if (str[x] == ' ')
            {
                bigtop += top;
                sumArr[pos++] = top;
                top = 0;
            }
            else
 
               
                top += str[x];
        }
 
        // son kelime toplam
        sumArr[pos] = top;
        bigtop += top;
        return bigtop;
    }
